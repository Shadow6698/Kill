-- Variáveis
local targetPlayer = nil
local rotating = false
local rotateSpeed = 300
local rotationConnection = nil

-- Função para girar grudado no player (altura dos pés)
local function startRotating(target)
	rotating = true
	local char = LocalPlayer.Character
	local targetChar = target.Character
	if not (char and targetChar) then return end

	rotationConnection = RunService.Heartbeat:Connect(function()
		if char and targetChar and targetChar:FindFirstChild("HumanoidRootPart") then
			local hrp = char:FindFirstChild("HumanoidRootPart")
			local targetHrp = targetChar.HumanoidRootPart
			if hrp then
				local angle = tick() * rotateSpeed
				local radius = 3
				local offset = CFrame.new(math.cos(angle)*radius, -2, math.sin(angle)*radius)
				hrp.CFrame = targetHrp.CFrame * offset * CFrame.Angles(0, math.rad(angle), 0)
			end
		end
	end)
end

-- Parar rotação
local function stopRotating()
	rotating = false
	if rotationConnection then
		rotationConnection:Disconnect()
		rotationConnection = nil
	end
end

-- Buscar jogador pelo nome
local function getPlayerByName(name)
	for _, plr in pairs(Players:GetPlayers()) do
		if plr.Name:lower() == name:lower() then
			return plr
		end
	end
	return nil
end

-- Botão MATAR
KillButton.MouseButton1Click:Connect(function()
	local nick = NickBox.Text
	local plr = getPlayerByName(nick)
	if plr and plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
		targetPlayer = plr

		local myChar = LocalPlayer.Character
		if myChar and myChar:FindFirstChild("HumanoidRootPart") then
			myChar.HumanoidRootPart.CFrame = plr.Character.HumanoidRootPart.CFrame * CFrame.new(0, -2, -2)
		end

		startRotating(plr)
	else
		warn("Jogador não encontrado.")
	end
end)

-- Botão HITKILL
HitKillButton.MouseButton1Click:Connect(function()
	local myChar = LocalPlayer.Character
	if myChar and myChar:FindFirstChild("HumanoidRootPart") then
		stopRotating()

		local hrp = myChar.HumanoidRootPart
		local oldCFrame = hrp.CFrame -- guarda a posição antes de teletransportar

		hrp.CFrame = CFrame.new(400000, 400000, 400000)

		local camera = workspace.CurrentCamera
		camera.CameraSubject = myChar:FindFirstChild("Humanoid")
		camera.CameraType = Enum.CameraType.Custom

		-- Depois de 4 segundos, volta para a posição antiga
		delay(4, function()
			if hrp and hrp.Parent then
				hrp.CFrame = oldCFrame
			end
		end)
	end
end)
