-- Variáveis principais
local targetPlayer = nil
local rotating = false
local rotateSpeed = 300
local rotationConnection = nil

-- Plataforma
local PlataformaPos = Vector3.new(64456972, 209335824, 74136968) -- ✅ Coordenada atualizada

-- Coordenada mais longe para teleporte
local FarPosition = Vector3.new(1e6, 1e6, 1e6)

-- Função de rotação (Matar)
local function startRotating(target)
	rotating = true
	local char = LocalPlayer.Character
	local targetChar = target.Character
	if not (char and targetChar) then return end

	rotationConnection = RunService.Heartbeat:Connect(function()
		if char and targetChar and targetChar:FindFirstChild("HumanoidRootPart") then
			local hrp = char:FindFirstChild("HumanoidRootPart")
			local targetHrp = targetChar.HumanoidRootPart
			if hrp then
				local angle = tick() * rotateSpeed
				local radius = 5
				hrp.CFrame = targetHrp.CFrame * CFrame.new(math.cos(angle)*radius, 0, math.sin(angle)*radius) * CFrame.Angles(0, math.rad(angle), 0)
			end
		end
	end)
end

-- Parar rotação
local function stopRotating()
	rotating = false
	if rotationConnection then
		rotationConnection:Disconnect()
		rotationConnection = nil
	end
end

-- Função para travar a câmera (de lado)
local function lockCameraToCharacterSideView(character)
	local camera = workspace.CurrentCamera
	camera.CameraSubject = character.Humanoid
	camera.CameraType = Enum.CameraType.Attach
end

-- Função para liberar câmera ao modo normal
local function unlockCamera()
	local camera = workspace.CurrentCamera
	camera.CameraType = Enum.CameraType.Custom
	camera.CameraSubject = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("Humanoid") or nil
end

-- Função para girar o personagem com força
local function spinCharacter(character, duration, spinForce)
	local hrp = character:FindFirstChild("HumanoidRootPart")
	if not hrp then return end

	local bav = Instance.new("BodyAngularVelocity")
	bav.AngularVelocity = Vector3.new(0, spinForce, 0) -- girar no eixo Y
	bav.MaxTorque = Vector3.new(0, math.huge, 0)
	bav.P = 10000
	bav.Parent = hrp

	task.delay(duration, function()
		bav:Destroy()
	end)
end

-- Função HitKill atualizada
local function hitKill()
	stopRotating()
	local char = LocalPlayer.Character
	if not char or not char:FindFirstChild("HumanoidRootPart") then return end

	local oldPos = char.HumanoidRootPart.CFrame

	-- Teleportar para coordenada longe
	char.HumanoidRootPart.CFrame = CFrame.new(FarPosition)

	-- Travar câmera com visão lateral
	lockCameraToCharacterSideView(char)

	-- Esperar o "chegar"
	task.wait(3)

	-- Girar o personagem com força 3000 por 5 segundos
	spinCharacter(char, 5, 3000)

	-- Esperar o fim da rotação
	task.wait(5)

	-- Destravar câmera
	unlockCamera()

	-- Voltar para posição antiga
	char.HumanoidRootPart.CFrame = oldPos
end

-- Função para buscar jogador pelo nick
local function getPlayerByName(name)
	for _, plr in pairs(Players:GetPlayers()) do
		if plr.Name:lower() == name:lower() then
			return plr
		end
	end
	return nil
end

-- Botões
KillButton.MouseButton1Click:Connect(function()
	local nick = NickBox.Text
	local plr = getPlayerByName(nick)
	if plr then
		targetPlayer = plr
		startRotating(plr)
	else
		warn("Jogador não encontrado.")
	end
end)

HitKillButton.MouseButton1Click:Connect(function()
	local nick = NickBox.Text
	local plr = getPlayerByName(nick)
	if plr then
		targetPlayer = plr
		hitKill()
	else
		warn("Jogador não encontrado.")
	end
end)
